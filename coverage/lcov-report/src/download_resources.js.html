// Handler for requesting tiles from different online sources
export const requestOnlineTiles = (
  style,
  apiKey,
  mapboxStyle,
  monthYear,
  bounds,
  minZoom,
  maxZoom,
  tempDir,
) => {
  return new Promise(async (resolve, reject) => {
    try {
      await downloadOnlineTiles(
        style,
        apiKey,
        mapboxStyle,
        monthYear,
        bounds,
        minZoom,
        maxZoom,
        tempDir,
      );
      resolve();
    } catch (error) {
      reject(error);
    }
  });
